1.接下来的程序是书上的例题

主要注意string，字符串，empty()判断是否为空，其次是string[0]；可以通过索引来找到数组的元素

` 

#include<iostream>
#include<string>
 #include<cctype>
using namespace std;
int main()
{
    string rsp;
    do{
        cout<<"please enter two values"<<endl;
        int val1=0,val2=0;
        cin>>val1>>val2;
        cout<<"The sum of"<<val1<<"and"<<val2
            <<"="<<val1+val2<<endl
            <<"more？ enter yes or no:";
        cin>>rsp;
    }while(!rsp.empty()&&rsp[0]!='n');
    return  0;
}

`
